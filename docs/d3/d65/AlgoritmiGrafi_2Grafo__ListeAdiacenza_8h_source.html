27 class liste_adiacenza {
28 public:
29  __gnu_cxx::hash_map< vertice *, std::list< adiacente >, hash_v > m;
30 
31 public:
32  liste_adiacenza() {};
33  ~liste_adiacenza() {};
34 
35  void inserisciVertice(vertice *v) { m[v] = std::list<adiacente>(); }
36  void eliminaVertice(vertice *v) {
37  m.erase(m.find(v));
38 
39  __gnu_cxx::hash_map< vertice *, std::list< adiacente >, hash_v >::iterator v_it = m.begin();
40  __gnu_cxx::hash_map< vertice *, std::list< adiacente >, hash_v >::iterator v_end = m.end();
41  std::list< adiacente >::iterator adj_it, adj_end, temp_it;
42  for (; v_it != v_end; ++v_it) {
43  adj_it = v_it->second.begin();
44  adj_end = v_it->second.end();
45  for (; adj_it != adj_end; ++adj_it)
46  if (adj_it->vertex == v) {
47  temp_it = adj_it;
48  ++temp_it;
49  v_it->second.erase(adj_it);
50  adj_it = temp_it;
51  }
52  }
53  }
54  void inserisciArco(vertice *x, vertice *y, int c) { m[x].push_back(adiacente(y, c)); }
55  void eliminaArco(vertice *x, vertice *y, int c) {
56  std::list< adiacente >::iterator it, end;
57  it = m[x].begin();
58  end = m[x].end();
59  for (; it != end; ++it)
60  if (it->vertex == y && it->costo == c)
61  { m[x].erase(it); break; }
62  }
63  void reset() { m.clear(); }
64  bool sonoAdiacenti(vertice *x, vertice *y) const {
65  std::list< adiacente >::const_iterator it = m.find(x)->second.begin();
66  std::list< adiacente >::const_iterator end = m.find(x)->second.end();
67  for (; it != end; ++it)
68  if (it->vertex == y)
69  return true;
70  return false;
71  }
72  unsigned contaOccorrenze(vertice *v) const {
73  unsigned g(0);
74 
75  __gnu_cxx::hash_map< vertice *, std::list< adiacente >, hash_v >::const_iterator v_it = m.begin();
76  __gnu_cxx::hash_map< vertice *, std::list< adiacente >, hash_v >::const_iterator v_end = m.end();
77  std::list< adiacente >::const_iterator adj_it, adj_end;
78  for (; v_it != v_end; ++v_it) {
79  adj_it = v_it->second.begin();
80  adj_end = v_it->second.end();
81  for (; adj_it != adj_end; ++adj_it)
82  if (adj_it->vertex == v)
83  ++g;
84  }
85 
86  return g;
87  }
88  unsigned dimensioneLista(vertice *v) const {
89  return m.find(v)->second.size();
90  }
