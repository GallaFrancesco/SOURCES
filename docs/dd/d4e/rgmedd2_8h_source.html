179 enum class BoundGuessPolicy {
180  M0_LINEAR, // each variable starts at its initial marking value or 1, then underestimated variables increase by their max_var_incr times restarts
181  M0_EXP, // each variable starts at its initial marking value or 1, then underestimated variables increase by their max_var_incr times 2^restarts
182  MAX0_LINEAR, // all variables start with the max token count in m0, then underestimated variables increase by their max_var_incr times restarts
183  MAX0_EXP, // all variables start with the max token count in m0, then underestimated variables increase by their max_var_incr times 2^restarts
184  M0_MAX0_EXP, // each variable starts at its initial marking value or 1, underestimated variables start with the max token count in m0,
185  // then underestimated variables increase by their max_var_incr times 2^(restarts-1)
186  LOAD_FROM_FILE // loads the bounds from the bnd file; if a bound of a place is not available m0 or 1 is used.
187  // The underestimated variables increase by their max_var_incr times 2^restarts
