57  std::string action;

60 vertice_array<struct rew_act> BellmanFordMin(const Grafo &G, vertice *s) {
61  unsigned n = G.numVertici();
62  vertice *v, *u;
63  arco *e;
64  double w;
65 
66  /* inizializza D tale che Dsv = +infinito per v != s, e Dss = 0 */
67  vertice_array<struct rew_act> Ds;
68  for_each_vertice(v, G) Ds[v].reward = INFINITO;
69  Ds[s].reward = 0;
70  Ds[s].action = "";
71  for (unsigned i = 1; i <= n; ++i) {
72  /* for each ( (u,v) in E ) do */
73  for_each_arco(e, G) {
74  u = e->sorg; v = e->dest; w = e->costo; // e = (u,v) con costo w
75  //cout<<"Nome:"<<info[];
76  if (Ds[u].reward != INFINITO) {
77  if (Ds[u].reward + w < Ds[v].reward) Ds[v].reward = Ds[u].reward + w; // rilassamento
78  if (Ds[u].action != "")
79  Ds[v].action = Ds[u].action + "," + info1[e];
80  else
81  Ds[v].action = " " + info1[e];
82  }
83  }
84  }
85  for_each_arco(e, G) {
86  u = e->sorg;
87  v = e->dest;
88  if (Ds[v].reward > Ds[u].reward + e->costo) {
89  cerr << "Graph contains a negative-weight cycle\n\n";
90  exit(1);
91  }
92  }
93  return Ds;
94 }

96 vertice_array <struct rew_act>BellmanFordMax(const Grafo &G, vertice *s) {
97  unsigned n = G.numVertici();
98  vertice *v, *u;
99  arco *e;
100  double w;
101  /* inizializza D tale che Dsv = +infinito per v != s, e Dss = 0 */
102  vertice_array<struct rew_act> Ds;
103  for_each_vertice(v, G) Ds[v].reward = (INFINITO) * -1;
104  Ds[s].reward = 0;
105  Ds[s].action = "";
106  for (unsigned i = 1; i <= n; ++i) {
107  for_each_arco(e, G) {
108  u = e->sorg; v = e->dest; w = e->costo; // e = (u,v) con costo w
109  if (Ds[u].reward != (INFINITO) * -1) {
110  if (Ds[u].reward + w > Ds[v].reward) Ds[v].reward = Ds[u].reward + w; // rilassamento
111  if (Ds[u].action != "")
112  Ds[v].action = Ds[u].action + "," + info1[e];
113  else
114  Ds[v].action = " " + info1[e];
115  }
116  }
117  }
118  for_each_arco(e, G) {
119  u = e->sorg;
120  v = e->dest;
121  if (Ds[v].reward < Ds[u].reward + e->costo) {
122  cerr << "Graph contains a positive-weight cycle\n\n";
123  exit(1);
124  }
125  }
126  return Ds;
127 }
