55 ILU0(Matrix &M, VerboseLevel verboseLvl) {
56  typedef typename Matrix::value_type value_type;
57  typedef typename Matrix::size_type size_type;
58  typedef typename ublas::matrix_row<Matrix>::iterator row_iterator, col_iterator;
59  value_type det = 1;
60  const size_type N = M.size1();
61 
62  if (verboseLvl >= VL_BASIC)
63  cout << "BUILDING ILU(0) PRECONDITIONER... " << flush;
64 
65  Matrix trM; // transpose structure of M
66  FastTranspose(M, trM);
67 
68  for (size_t r = 0; r < N - 1; r++) {
69  value_type d = safe_inv(M(r, r));
70  det *= M(r, r);
71 
72  ublas::matrix_row<Matrix> rth_col(trM, r);
73  col_iterator it_r(rth_col.begin()), endIt_r(rth_col.end());
74  for (; it_r != endIt_r; ++it_r) {
75  const size_type i = it_r.index();
76  if (i <= r)
77  continue;
78  if (M(i, r) == 0)
79  continue;
80  value_type e = M(i, r) * d;
81  M(i, r) = e;
82 
83  ublas::matrix_row<Matrix> ith_row(M, i);
84  row_iterator M_ij(ith_row.begin()), endM_ij(ith_row.end());
85  for (; M_ij != endM_ij; ++M_ij) {
86  const size_type j = M_ij.index();
87  if (j <= r)
88  continue;
89  if (*M_ij == 0 || M(r, j) == 0)
90  continue;
91  *M_ij -= e * M(r, j);
92  }
93  }
94  }
95  if (verboseLvl >= VL_BASIC) {
96  cout << "BUILT: det=" << det << ", nnz=" << M.nnz();
97  double fill_in = int(double(M.nnz()) / (M.size1() * M.size2()) * 10000) * 0.01;
98  cout << ", fill-in=" << fill_in << "%." << endl;
99  if (verboseLvl >= VL_VERBOSE && M.size1() < 50) {
100  print_matrix(cout, M, "M");
101  cout << endl << endl;
102  }
103  }
104  return det;
105 }

172  else {

177  while (U_kj != U_kend) {

180  ++U_kj;
181  }
182  }

184  }

193  if (std::abs(diag_i) > row_threshold)

195  else

206  if (wIt.index() < i)
207  qL.push(make_pair(*wIt, wIt.index()));
208  else if (wIt.index() > i)
209  qU.push(make_pair(*wIt, wIt.index()));
210  }
211  ++wIt;
212  }

214  size_t numL = 0, numU = 0;
215  while (numL < K && !qL.empty()) {
216  wL(qL.top().second) = qL.top().first;
217  qL.pop();
218  numL++;
219  }
220  while (numU < K && !qU.empty()) {
221  wU(qU.top().second) = qU.top().first;
222  qU.pop();

232  vec_iterator_t wU_j = wU.begin(), wU_jend = wU.end();
233  for (; wU_j != wU_jend; ++wU_j)
