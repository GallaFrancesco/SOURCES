86 template<class VecIter, class Func>
87 std::function<ostream& (ostream &)>
88 print_vec(VecIter first, VecIter last, Func func,
89  const vec_out_style_t &fmt = s_pvfmtBracketsSpaced) {
90  return [ = ](ostream & os) -> ostream& {
91  auto it = first;
92  os << fmt.openPar;
93  for (size_t n = 0; it != last; ++it, ++n) {
94  os << (n == 0 ? fmt.emptySpc : fmt.sep);
95  os << func(*it);
96  }
97  return os << fmt.emptySpc << fmt.closePar;
98  };
99 }

175 struct simple_tokenizer {
176  inline simple_tokenizer(istream *_is, const skip_spaces_fn _sfn = skip_spaces)
177  : is(_is), skip_sp(_sfn) { }
178 
179  // Read an exact token, throw an exception if the token is different
180  simple_tokenizer &readtok(const char *token);
181 
182  // Extract a token (and return it) using a simple char-based rule
183  simple_tokenizer &gettok(string &buf, int(*lexerRule)(int) = isnotspace,
184  size_t maxChars = std::numeric_limits<size_t>::max());
185 
186  // Read a simple token (int, double, string, ...) from the current row
187  template<typename T>
188  inline simple_tokenizer &read(T &obj) {
189  (*is) >> obj;
190  ncol += is->gcount();
191  return *this;
192  }
193 
194  // skip n lines
195  simple_tokenizer &skiplines(size_t numLines = 1);
196 
197  // read a single line (without the '\n') and store it into line,
198  // drop the '\n' from the stream - no space skipping is done.
199  simple_tokenizer &getline(string &line);
200 
201  inline bool get(int &c) {
202  c = is->get();
203  if (c == '\n') { nrow++; ncol = 0; }
204  else { ncol++; }
205  return is->good();
206  }
207 
208  inline int peek() { return is->peek(); }
209 
210  inline bool unget() { ncol--; is->unget(); return is->good(); }
211 
212  // Format a string with the current row/column position
213  string pos() const;
214 
215  inline operator bool() { return (is->good()); }
216 
217 private:
218  istream *is;
219  size_t nrow = 0, ncol = 0;
220  skip_spaces_fn skip_sp;
221 
222  friend ostream &operator << (ostream &os, const simple_tokenizer &st);
223 };
