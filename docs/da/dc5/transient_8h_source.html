54 void StandardUniformization(const Matrix &P, double lambda,
55  const Vector1 &pi0, Vector2 *pi, Vector2 *pi_acc,
56  const double t, const double epsilon,
57  const RowSelector &rows, const ColSelector &cols,
58  const VerboseLevel verboseLvl) {
59  const size_t N = P.size1();
60  const bool compute_accum = (pi_acc != nullptr); // Cumulative
61  const bool compute_inst = (pi != nullptr); // Instantaneous
62  assert(N > 1 && P.size2() == N && pi0.size() == N && t > 0.0);
63  assert(compute_accum || compute_inst);
64 
65  // Number of uniformization steps M: solution of pi(0) => pi(t) is divided
66  // into M steps: pi(0) => pi(t/M) => pi(2*t/M) => ... => pi(t)
67  double MAX_LAMBDA_T = 290.0;
68  const size_t M = 1 + size_t(floor((lambda * t) / MAX_LAMBDA_T));
69  const double tM = t / M; // tm is the time step
70 
71  // Find the number R of iterations needed to reach the required accuracy
72  // R is the right truncation point of the Taylor series.
73  size_t R = 0;
74  double sigma = 1;
75  double xi = 1;
76  double eta = (1 - epsilon / M) / exp(-lambda * tM);
77  while (sigma < eta) {
78  R++;
79  xi *= (lambda * tM) / R;
80  sigma += xi;
81  }
82 
83  if (verboseLvl >= VL_BASIC) {
84  cout << "STANDARD UNIFORMIZATION(t=" << t << ", Lambda=" << lambda << "):";
85  cout << " M=" << M << " steps, R=" << R << " right truncation point, ";
86  cout << (M * R) << " vector*matrix products." << endl;
87  }
88 
89  // The uniformization is done in M steps: each of these step advances
90  // the probability vector pi of (t/M) time units.
91  ublas::zero_vector<double> zero_vec(N);
92  const double expLambdaT = exp(-lambda * tM);
93  Vector2 y, yTmp(N), z, pi_t;
94 
95  if (!compute_inst) {
96  // We need the pi vector in any case, because it's used to initialize
97  // y at each new step iteration; however, we will skip the last m step
98  pi = &pi_t;
99  }
100  (*pi) = pi0;
101  if (compute_accum)
102  (*pi_acc) = zero_vec;
103 
104  for (size_t m = 0; m < M; m++) {
105  const bool last_m_iter = (m == M - 1);
106  // To approximate pi(t/M), sum up the first R terms of:
107  // pi(tM) = SUM(k=1..R){ y_k * beta_k }
108  // with y_k = pi0 * P^k k-th exponentiation of pi0 * P
109  // beta_k = e^(-lambda*tM) * ((lambda*tM)^k)/k! Poisson value.
110  double beta = 1.0, gamma = (1 - expLambdaT);
111  copy_vec(y, *pi, rows); // y = (*pi)
112 
113  // Cumulative probabilities follows the formula:
114  // pi_acc(tM) = 1/lambda * SUM(k=0..R){ y_k * gamma_k }
115  // with gamma_k = complement of the k-th cumulative Poisson value.
116  if (compute_accum) {
117  copy_vec(z, *pi, rows);
118  mult_vec(z, 1 - expLambdaT, rows); // z = (*pi) * (1 - expLambdaT)
119  }
120 
121  for (size_t k = 1; k <= R; k++) {
122  // y is the probability vector after k steps of the DTMC P
123  prod_vecmat(yTmp, y, P, rows, cols); // y' = prod(y, P)
124  copy_vec(y, yTmp, rows); // y = y';
125  // beta(k, lambda*tM) Poisson distribution function
126  beta *= ((lambda * tM) / k);
127 
128  // Sum up the Taylor series terms
129  if (compute_inst || !last_m_iter)
130  plus_assign_mult(*pi, beta, y, rows); // pi += beta * y
131  if (compute_accum) {
132  gamma -= beta * expLambdaT;
133  plus_assign_mult(z, gamma, y, rows); // z += gamma * y
134  }
135  }
136 
137  if (compute_inst || !last_m_iter)
138  mult_vec(*pi, expLambdaT, rows); // pi *= exp(-lambda*tM)
139  if (compute_accum)
140  plus_assign_mult(*pi_acc, 1 / (lambda * t), z, rows); // pi_acc += z / lambda
141  }
142 }

169  const Vector1 &pi0, Vector2 *pi, Vector2 *pi_acc,

175  const bool compute_inst = (pi != nullptr); // Instantaneous
176  assert(N > 1 && P.size2() == N && pi0.size() == N && t > 0.0);
177  assert(compute_accum || compute_inst);
178 
179  // Compute Poisson weights W in the relevant interval [L, R]
180  FoxGlynnWeights fgw;
181  if (!FG_Weighter(lambda * t, epsilon, 1.0e-300, 1.0e300, fgw, verboseLvl))
182  return false;
183 
184  if (verboseLvl >= VL_BASIC) {
185  cout << "FOX-GLYNN UNIFORMIZATION(t=" << t << ", Lambda=" << lambda;
186  cout << ", L=" << fgw.L << ", R=" << fgw.R << ", SumW=" << fgw.SumW << ")";
187  cout << endl;
188  }
189 
190  // Difference between 2 magnitude orders for multiorder_vector
191  const int ORD_DIFF = 16;
192  const double minVal = min(fgw.W[0], fgw.W[fgw.W.size() - 1]) / fgw.SumW * 1.0e-15;
193  multiorder_vector mo_pi(N, 1.0, minVal, ORD_DIFF);
194  multiorder_vector mo_pi_acc(N, 1.0, minVal, ORD_DIFF);
195 
196  // Fox-Glynn Uniformization uses a modified Taylor series with both left
197  // and right truncation, in accordance with the formula:
198  // pi(t) = SUM{n=L..R}( Phi(n) * (W(n)/SumW) )
199  // with: Phi(n) = pi0 * (P^n) n-th vector exponentiation
200  ublas::vector<double> Phi; // Prob. distribution after n transitions
201  double Gamma = 1.0; // Complementary cumulative probability
202  Phi = pi0;
203 
204  // Compute the first L vector*matrix products
205  ublas::vector<double> Phi_prime;
206  for (int n = 0; n < fgw.L; n++) {
207  if (compute_accum) {
208  for (size_t i = 0; i < rows.count(); i++)
209  mo_pi_acc.add(rows[i], Phi(rows[i]));
210  }
211  prod_vecmat(Phi_prime, Phi, P, rows, cols);
212  Phi_prime.swap(Phi);
213  }
214  // Compute the [L..R] range
215  for (int n = fgw.L; n < fgw.R; n++) {
216  double beta = (fgw.W[n - fgw.L] / fgw.SumW);
217  if (compute_accum) {
218  Gamma -= beta;
219  for (size_t i = 0; i < rows.count(); i++)
220  mo_pi_acc.add(rows[i], Gamma * Phi(rows[i]));
221  }
222  if (compute_inst) {
223  for (size_t i = 0; i < rows.count(); i++)
224  mo_pi.add(rows[i], beta * Phi(rows[i]));
225  }
226 
227  if (n != fgw.R - 1) {
228  //Phi' = prod(Phi, P);
229  prod_vecmat(Phi_prime, Phi, P, rows, cols);
230  Phi_prime.swap(Phi);
231  }
232  }
233 
234  // Pack the multiorder solution vectors into *pi and *pi_acc
235  if (compute_accum) {
236  // Add the [0..L-1] probabilities for last, to avoid roundoffs
237  mo_pi_acc.pack(*pi_acc); //TODO: pack() sugli elementi in rows
238  // Normalize the cumulative probabilities
239  mult_vec(*pi_acc, 1 / (lambda * t), rows);
240  }
241  if (compute_inst) {
242  mo_pi.pack(*pi); //TODO: pack() sugli elementi in rows
243  }
244 
245  return true;
246 }
