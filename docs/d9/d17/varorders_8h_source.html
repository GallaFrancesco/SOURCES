130 // Basic accumulator of statistics measures
131 class accumulator {
132  double m_sum = 0, m_sum2 = 2;
133  double m_min = 0, m_max = 0;
134  size_t m_count = 0;
135 public:
136  accumulator() {}
137  accumulator(const accumulator&) = default;
138  accumulator(accumulator&&) = default;
139 
140  accumulator& operator = (const accumulator&) = default;
141  accumulator& operator = (accumulator&&) = default;
142 
143  inline void operator += (double value) {
144  m_sum += value;
145  m_sum2 += (value * value);
146  if (count() == 0)
147  m_min = m_max = value;
148  m_min = std::min(m_min, value);
149  m_max = std::max(m_max, value);
150  m_count++;
151  }
152 
153  inline double count() const { return m_count; }
154  inline double min() const { return m_min; }
155  inline double max() const { return m_max; }
156  inline double sum() const { return m_sum; }
157 
158  inline double mean() const { return m_count ? (m_sum / m_count) : 0; }
159  inline double mean2() const { return m_count ? (m_sum2 / m_count) : 0; }
160  inline double variance() const {
161  double m = mean();
162  return m == 0 ? 0 : mean2() / (m * m);
163  }
164  inline double stddev() const { return sqrt(variance()); }
165  inline double root_mean_sq() const { return m_count ? sqrt(m_sum2 / m_count) : 0; }
166 };
