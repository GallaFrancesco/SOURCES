28 class EmcVectorMatrixProduct : boost::noncopyable {
29 public:

31  const MRPSubsetList &_Subsets,
32  KolmogorovEquationDirection _ked,
33  const RowSelector &_rows, const ColSelector &_cols)

44  // Compute a vector * matrix product
45  ublas::vector<double> product(const ublas::vector<double> &inVec) const {
46  const marking_t N_MRP(mrp.N());
47  ublas::vector<double> out(N_MRP);
48  fill(out.begin(), out.end(), 0.0);
49  NullIterPrintOut printOut;
50  MRPSubsetList::selector ExpStates(Subsets.selectorOf(EXP_SUBSET));
51  vector<GenMarkingSubset>::const_iterator SgenIt;
52 
53  if (ked == KED_FORWARD) {
54  //=================== Forward: y = x.P ======================
55 
56  // The implicit EMC x Vector product is done in two phases.
57  // 1) Exponential states:
58  // input vector is multiplied with P^E = I^E - diag^{-1}(Q^E)Q^E
59  prod_vec_invDiagM_M(out, inVec, mrp.Q, ExpStates, cols);
60  add_vec(out, inVec, ExpStates);
61 
62  // 2) Deterministic states:
63  // input vector is multiplied with Omega.Delta + Psi.Qbar
64  ublas::vector<double> ag(N_MRP), bg(N_MRP);
65  for (SgenIt = mrp.Gen.begin(); SgenIt != mrp.Gen.end(); ++SgenIt) {
66  if (Subsets.numStatesIn(SgenIt->index) == marking_t(0))
67  continue;
68  MRPSubsetList::selector GenStates(Subsets.selectorOf(SgenIt->index));
69 
70  // Compute uniformization over the Sgen subset *SgenIt
71  // TODO: note that here the spar.epsilon precision is used in the
72  // uniformization. In some models, this epsilon needs to be
73  // lowered (for instance: spar.epsilon * 1.0e-1) in order to
74  // converge.
75  UniformizationGen(mrp.Q, inVec, ag, &bg, SgenIt->fg.c_str(),
76  spar.epsilon * 1.0e-1, KED_FORWARD,
77  GenStates, cols, printOut);
78  add_prod_vecmat(out, ag, mrp.Delta, GenStates, cols);
79  add_prod_vecmat(out, bg, mrp.Qbar, GenStates, cols);
80  }
81  }
82  else {
83  //================== Backward: y = P.x ======================
84  prod_invDiagM_M_vec(out, mrp.Q, inVec, ExpStates, cols);
85  add_vec(out, inVec, ExpStates);
86 
87  ublas::vector<double> DeltaX(N_MRP), QbarX(N_MRP), outExp, outExpInt;
88  ublas::vector<double> *no_vec = nullptr;
89  for (SgenIt = mrp.Gen.begin(); SgenIt != mrp.Gen.end(); ++SgenIt) {
90  if (Subsets.numStatesIn(SgenIt->index) == marking_t(0))
91  continue;
92  MRPSubsetList::selector GenStates(Subsets.selectorOf(SgenIt->index));
93 
94  prod_matvec(DeltaX, mrp.Delta, inVec, GenStates, cols);
95  UniformizationGen(mrp.Q, DeltaX, outExp, no_vec, SgenIt->fg.c_str(),
96  spar.epsilon * 1.0e-1, KED_BACKWARD,
97  GenStates, cols, printOut);
98  add_vec(out, outExp, cols);
99 
100  prod_matvec(QbarX, mrp.Qbar, inVec, GenStates, cols);
101  if (vec_sum(QbarX) != 0) {
102  UniformizationGen(mrp.Q, QbarX, outExp, &outExpInt, SgenIt->fg.c_str(),
103  spar.epsilon * 1.0e-1, KED_BACKWARD,
104  GenStates, cols, printOut);
105  add_vec(out, outExpInt, cols);
106  }
107  }
108  }
109 
110  productCount++;
111  return out;
112  }
113 
114 private:

117  const KolmogorovEquationDirection ked;
118  const RowSelector &rows;
119  const ColSelector &cols;
120  const MRPSubsetList &Subsets;
121  mutable size_t productCount;
122 };
