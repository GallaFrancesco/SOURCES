121 Inserimento di un nuovo elemento.

123 template<class Elem, class Chiave>
124 nodo_binomiale<Elem, Chiave> *HeapBinomialeRilassato<Elem, Chiave>::insert(const Elem &e,
125  const Chiave &k) {
126  /* aggiunge alla foresta un nuovo B0 con dati 'e' e 'k' */
127  nodo_binomiale<Elem, Chiave> *v = new nodo_binomiale<Elem, Chiave>(e, k);
128  (this->foresta).push_back(v);
129  ++(this->itemsCount);

131  aggiornaMin();

133  (this->lista_nodi).push_back(v);

166 /*
167 Fusione di due code di priorita' in un'unica coda di priorita'.
168 */
169 template<class Elem, class Chiave>
170 void HeapBinomialeRilassato<Elem, Chiave>::merge(const HeapBinomialeRilassato &c1,
171  const HeapBinomialeRilassato &c2) {
172  /* unisci le liste delle radici dei due heap binomiali */
173  for (foresta_const_iterator it = (c1.foresta).begin(); it != (c1.foresta).end(); ++it)
174  (this->foresta).push_back(*it);
175  for (foresta_const_iterator it = (c2.foresta).begin(); it != (c2.foresta).end(); ++it)
176  (this->foresta).push_back(*it);
177 
178  this->itemsCount += (c1.itemsCount + c2.itemsCount);
179 
180  aggiornaMin();
181 }

187 /*
188 Aggiorna 'min'.
189 */
190 template<class Elem, class Chiave>
191 void HeapBinomialeRilassato<Elem, Chiave>::aggiornaMin() {
192  nodo_binomiale<Elem, Chiave> *r;
193 
194  min = (this->foresta).front();
195 
196  for_each_radice(r) {
197  if (r->chiave < min->chiave) {
198  min = r;
199  }
200  }
