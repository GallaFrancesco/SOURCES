29 struct sz_tuple_t {

31  location_t _z = location_t(size_t(-1)))
32  : s(_s), z(_z) { }
33 
34  marking_t s; // The CTMC state

36 
37  inline bool operator < (const sz_tuple_t &t) const {
38  return (s < t.s || (s == t.s && z < t.z));
39  }
40  inline bool operator == (const sz_tuple_t &t) const {
41  return (s == t.s && z == t.z);
42  }
43  inline bool isTop() const;
