153 static inline
154 bool matrix_row_has_nonzeroes(const ublas::compressed_matrix<double> &M,
155  const size_t row, bool &isAbsorbing) {
156  matRow_t Mrow(M, row);
157  rowIter_t M_ij(Mrow.begin()), M_rowEnd(Mrow.end());
158  bool hasNonZeroDiag = false;
159  while (M_ij != M_rowEnd) {
160  const size_t col = M_ij.index();
161  if (*M_ij != 0.0) {
162  if (col != row) {
163  isAbsorbing = false;
164  return true;
165  }
166  hasNonZeroDiag = true;
167  }
168  ++M_ij;
169  }
170  return hasNonZeroDiag;
171 }
