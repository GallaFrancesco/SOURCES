107 struct rgstate_t {

116  inline size_t index() const { assert(num != 0); return abs(num) - 1; }

122  { num = (type == TANGIBLE ? ssize_t(index) + 1 : -ssize_t(index) - 1); }

125  inline bool operator!= (const rgstate_t e) const { return num != e.num; }
126  inline bool operator< (const rgstate_t e) const { return num < e.num; }
