63 sccindex_t TarjanIterative(const Matrix &mat, std::vector<sccindex_t> &SCC) {
64  const size_t N = mat.size1();
65  assert(mat.size2() == N);
66  int IndexCount = 0;
67  sccindex_t SccCount(0);
68  std::vector<scc_node> Nodes(N);
69  std::vector<int> NodeStack;
70  std::vector<dfs_visit_node> DepthFirstStack;
71  NodeStack.reserve(N);
72  DepthFirstStack.reserve(N);
73  SCC.resize(N);
74  std::fill(SCC.begin(), SCC.end(), -1);
75 
76  for (size_t startNode = 0; startNode < N; startNode++) {
77  if (Nodes[startNode].index >= 0)
78  continue; // already visited
79 
80  dfs_visit_node start(startNode, DFS_ENTER);
81  DepthFirstStack.push_back(start);
82 
83  // Start the Depth-First visit from the top-of-stack node
84  while (!DepthFirstStack.empty()) {
85  dfs_visit_node &cn = DepthFirstStack.back();
86 
87  switch (cn.visitState) {
88  case DFS_ENTER:
89  // Start visiting node cn.v, pushing it onto the node stack
90  // Set depth index of node v and push it on the stack
91  Nodes[cn.v].index = Nodes[cn.v].lowlink = IndexCount;
92  IndexCount += 1;
93  NodeStack.push_back(cn.v);
94  Nodes[cn.v].on_stack = true;
95  cn.visitState = DFS_CONTINUE;
96  break;
97 
98  case DFS_CONTINUE:
99  // Resume the previously stopped visit at node cn.w
100  Nodes[cn.v].lowlink = std::min(Nodes[cn.v].lowlink,
101  Nodes[cn.w].lowlink);
102  break;
103  }
104 
105  // Visit all the successors of node cn.v , from where we
106  // stopped (after the first cn.j successors).
107  typedef typename ublas::matrix_row<const Matrix> row_t;
108  typedef typename row_t::const_iterator row_iter_t;
109 
110  row_t row(mat, cn.v);
111  row_iter_t iter = row.find(cn.j); // Resume at the j-th non-zero edge
112 
113  // Visit the successors of v
114  bool visit_other_nodes_before = false;
115  while (iter != row.end()) {
116  cn.w = iter.index();
117  double edge_value = *iter;
118  iter++;
119  cn.j++;
120  if (cn.v == cn.w)
121  continue;
122  // Ignore edges mat(cn.v, cn.w) == 0
123  if (edge_value == 0.0) // FIXME: abs(edge_value) <= threshold
124  continue;
125

127  if (Nodes[cn.w].index < 0) {
128  dfs_visit_node visit_w(cn.w, DFS_ENTER);
129  DepthFirstStack.push_back(visit_w);
130  visit_other_nodes_before = true;
131  break;
132  }
133  else if (Nodes[cn.w].on_stack) { // Is w onto the stack?
134  Nodes[cn.v].lowlink = std::min(Nodes[cn.v].lowlink,
135  Nodes[cn.w].index);
136  }
137  }
138  // Check if we must visit other nodes before v successors
139  if (visit_other_nodes_before)
140  continue;
141 
142  // Test if v is an SCC root
143  if (Nodes[cn.v].index == Nodes[cn.v].lowlink) {
144  //cout << "SCC " << SccCount << ": ";

148  NodeStack.pop_back();

152  }
153  while (w != cn.v);

156  }

160  }

188 void BottomStronglyConnectedComponents(const Matrix &mat, BSCC &bscc) {
189  const size_t N = mat.size1();
190  assert(mat.size2() == N);
191 
192  // Get the SCCs
193  sccindex_t SccCount = TarjanIterative(mat, bscc.indices);
194 
195  if (SccCount == sccindex_t(1)) {
196  bscc.numBscc = 0; // Only one large BSCC, with no initial transient.
197  return;
198  }
199 
200  // Find bottom SCCs
201  vector<bool> IsBSCC(SccCount, true);
202  typename Matrix::const_iterator1 rowIt;
203  typename Matrix::const_iterator2 colIt;
204  for (rowIt = mat.begin1(); rowIt != mat.end1(); ++rowIt) {
205  size_t i = rowIt.index1();
206  for (colIt = rowIt.begin(); colIt != rowIt.end(); ++colIt) {
207  size_t j = colIt.index2();
208  if (bscc.indices[i] != bscc.indices[j])
209  IsBSCC[ bscc.indices[i] ] = false;
210  }
211  }
212 
213  // Remap SCC indexes into BSCC indexes
214  sccindex_t BsccCount = 0;
215  vector<sccindex_t> BsccRemap(SccCount, -1);
216  for (int i = 0; i < SccCount; i++)
217  if (IsBSCC[i])
218  BsccRemap[i] = BsccCount++;
219  for (size_t i = 0; i < N; i++)
220  bscc.indices[i] = BsccRemap[ bscc.indices[i] ];
221 
222  bscc.numBscc = BsccCount;
223 }
