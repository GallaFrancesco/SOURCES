101 template<class Elem, class Chiave>
102 void HeapFibonacci<Elem, Chiave>::decreaseKey(nodo_binomiale<Elem, Chiave> *v, const Chiave &d) {
103  v->chiave -= d;
104  if (v->chiave < (this->min)->chiave) this->min = v;
105 
106  if (v->padre) {
107  if (v->chiave < (v->padre)->chiave) staccaInCascata(v);
108  }
109 }

115 /*
116 Figura 8.11
117 Procedura 'staccaInCascata' per il ripristino delle proprieta'
118 di struttura e ordinamento in un heap di Fibonacci.
119 La procedura taglia l'arco tra 'v' e il suo genitore 'u'.
120 Se 'u' non e' radice e aveva gia' perso un figlio, stacca ricorsivamente anche 'u'
121 in modo che ogni nodo non-radice perda al piu' un figlio.
122 */
123 template<class Elem, class Chiave>
124 void HeapFibonacci<Elem, Chiave>::staccaInCascata(nodo_binomiale<Elem, Chiave> *v) {
125  /* u = padre(v) */
126  nodo_binomiale<Elem, Chiave> *u = v->padre;
127 
128  /* togli 'v' dalla lista dei figli di 'u' e aggiungi 'v' alla lista delle radici dell'heap. */
129  for (foresta_iterator it = (u->lista_figli).begin(); it != (u->lista_figli).end(); ++it) {
130  if (*it == v) {
131  (u->lista_figli).erase(it);
132  break;
133  }
134  }
135  v->padre = 0;
136  (this->foresta).push_back(v);
137 
138  v->lutto = false;
139  if (u->padre != 0) { // if( u non e' una radice ) then
140  if (u->lutto) staccaInCascata(u);
141  else u->lutto = true;
142  }
143 }
