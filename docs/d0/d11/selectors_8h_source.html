188  inline void ConstructSubsetList(extsubset_t _subsetNum,
189  vector<extsubset_t> &_stateToSubset) {
190  subsetNum = _subsetNum;
191  stateToSubset.swap(_stateToSubset);
192  stateToSubsetPos.resize(numStates());
193 
194  // Count the number of states in each subset
195  vector<size_t> stCounts(subsetNum);
196  for (size_t i = 0; i < stateToSubset.size(); i++) {
197  const subsetindex_t sid(conv.externalSubsetTo0BasedIndex(stateToSubset[i]));
198  assert(sid >= subsetindex_t(0) && sid < subsetindex_t(subsetCount()));
199  stateToSubsetPos[i] = stCounts[sid];
200  stCounts[sid]++;
201  }
202  subsetStart.resize(size_t(subsetNum) + 1);
203  subsetStart[0] = 0;
204  for (subsetindex_t s = subsetindex_t(0); s < subsetindex_t(subsetCount()); s++)
205  subsetStart[s + subsetindex_t(1)] = subsetStart[s] + stCounts[s];
206 
207  // Build up the subset->state list nthSubset
208  nthSubset.resize(stateToSubset.size());
209  fill(stCounts.begin(), stCounts.end(), 0);
210  for (extstate_t i = 0; i < extstate_t(stateToSubset.size()); i++) {
211  const subsetindex_t sid(conv.externalSubsetTo0BasedIndex(stateToSubset[i]));
212  assert(sid >= subsetindex_t(0) && sid < subsetindex_t(subsetCount()));
213  nthSubset[extstate_t(subsetStart[sid] + stCounts[sid])] = i;
214  ++stCounts[sid];
215  }
216  }
