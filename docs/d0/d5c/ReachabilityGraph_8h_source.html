139 inline size_t RG::countNumOfTimedPaths(bool expNoPreem, bool expPreem,
140  bool det) const {
141  vector<ReachableMarking>::const_iterator rmIt;
142  size_t cnt = 0;
143  for (rmIt = markSet.begin(); rmIt != markSet.end(); ++rmIt) {
144  bool hasDetEnab = (rmIt->enabledDetInds.size() > 0);
145  if (!hasDetEnab) {
146  // Only exponential arcs, no deterministic transitions
147  if (expNoPreem)
148  cnt += rmIt->timedPaths.size();
149  continue;
150  }
151  vector<RGArc>::const_iterator arcIt = rmIt->timedPaths.begin();
152  for (; arcIt != rmIt->timedPaths.end(); ++arcIt) {
153  if (arcIt->distrib == DET) {
154  if (det)
155  cnt++;
156  }
157  else if (arcIt->distrib == EXP) {
158  if (arcIt->preempted.size() == 0 && expNoPreem)
159  cnt++;
160  else if (arcIt->preempted.size() > 0 && expPreem)
161  cnt++;
162  }
163  }
164  }
165  return cnt;
166 }
